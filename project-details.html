Frontend Development


// Example of Property Listing Widget
class PropertyListingCard extends StatelessWidget {
  final Property property;
  
  const PropertyListingCard({Key? key, required this.property}) : super(key: key);
  
  @override
  Widget build(BuildContext context) {
    return Card(
      elevation: 4,
      child: Column(
        children: [
          CachedNetworkImage(
            imageUrl: property.imageUrl,
            placeholder: (context, url) => ShimmerLoading(),
          ),
          ListTile(
            title: Text(property.title),
            subtitle: Text(property.location),
            trailing: Text('\$${property.price}'),
          ),
          PropertyActions(property: property),
        ],
      ),
    );
  }
}

// Property State Management using Riverpod
final propertyProvider = StateNotifierProvider>((ref) {

class PropertyNotifier extends StateNotifier> {
  PropertyNotifier() : super([]);
  
  Future fetchProperties() async {
    final properties = await PropertyService.getProperties();
    state = properties;
  }
  
  void addProperty(Property property) {
    state = [...state, property];
  }
}
